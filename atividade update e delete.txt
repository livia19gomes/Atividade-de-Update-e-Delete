/*12 - Delete todas a receitas e seu itens que não estão relacionadas a nenhuma consulta.*/
DELETE FROM ITENSRECEITA i
WHERE i.ID_RECEITA IN 
(SELECT r.id_receita FROM RECEITA r 
WHERE r.ID_AGENDAMENTO NOT IN (SELECT a.id_agendamento FROM AGENDAMENTO a
WHERE a.id_agendamento = r.ID_AGENDAMENTO))

delete FROM RECEITA r
WHERE r.ID_AGENDAMENTO NOT IN 
(SELECT a.id_agendamento FROM AGENDAMENTO a
WHERE a.id_agendamento = r.ID_AGENDAMENTO)
